13/// 「歲」，相鄰兩冬至間的時段，或自冬至所在月（十一月）至下一冬至前月（十月或閏十月）的時段。
14///
15/// 支持的年份取決於曆表數據，見 [`ephemeris`]。
16///
17/// 注意：「歲」與「年」在曆法上不同，年以正月為首，但曆法編算須以兩冬至間的「歲」為基礎，本程序亦以「歲」編排，並依日期計算所在「年」。
18///
19/// # 用例
20///
21/// ```
22/// use kalendarilo::Date;
23/// use kalendarilo::chinese::{Annus, Month::*};
24///
25/// let date = Date::from_gregorian(2000, 1, 1).unwrap();
26/// let annus = Annus::from_date(date).unwrap();
27///
28/// assert_eq!(Ok((1999, Common(11), 25)), annus.ymd_for(date));
29/// ```
30#[derive(Debug, Clone)]
31pub struct Annus {
32    /// 序號，為該歲大部分時段所在的公元年
33    pub annus: i32,
34    /// 該歲的曆表
35    pub ephemeris: &'static ephemeris::Annus,
36    /// 全部月首，包括次一歲首月用以標記本歲最末日
37    pub months: Vec<NewMoon>,
38}
39/// 月首信息
40#[derive(Debug, Copy, Clone)]
41pub struct NewMoon {
42    /// 月名
43    pub month: Month,
44    /// 月首所在日期
45    pub date: Date,
46}
47/// 月名，`Common` 為平月，`Leap` 為閏月。
48#[derive(Debug, Copy, Clone, Eq, PartialEq, Hash)]
49pub enum Month {
50    Common(i32),
51    Leap(i32),
52}
53impl Month {
54    /// 取得月序號，無論平閏。
55    pub fn num(&self) -> i32 {
56        use Month::*;
57        *match self {
58            Common(v) | Leap(v) => v,
59        }
60    }
61    /// 閏月為 `true`，平月為 `false`
62    pub fn is_leap(&self) -> bool {
63        matches!(self, Self::Leap(_))
64    }
65    /// 取得月名的文本形式，十一、十二月稱「冬月」「臘月」。
66    pub fn name(&self) -> String {
67        fmt::month(*self)
68    }
69}

71impl Annus {
72    /// 取得與公元 `annus` 年對應的歲。
73    ///
74    /// 若曆表無該歲資料則返回 `None`。
75    ///
76    /// # 用例
77    ///
78    /// ```
79    /// use kalendarilo::chinese::Annus;
80    ///
81    /// let annus = Annus::new(2000).unwrap();
82    /// ```
83    pub fn new(annus: i32) -> Option<Self> {
84        use Month::*;
85
86        let ephemeris = ephemeris::Annus::get(annus)?;
87        let new_moon_dates: Vec<_> = ephemeris
88            .moon_phase
89            .iter()
90            .map(|arr| date_cst(arr[0]))
91            .collect();
92        let ws = date_cst(ephemeris.solar_term[0]);
93        let ws_next = date_cst(ephemeris.solar_term[24]);
94        let m11_idx = new_moon_dates.partition_point(|date| date <= &ws) - 1;
95        let m11n_idx = new_moon_dates.partition_point(|date| date < &ws_next) - 1;
96        let mut needs_leap = match m11n_idx - m11_idx {
97            12 => false,
98            13 => true,
99            _ => panic!("{} months between winter solstices", m11n_idx - m11_idx),
100        };
101
102        let mut months = Vec::with_capacity(m11n_idx - m11_idx);
103        let mut month = 10;
104        let mut term = 0;
105        for i in m11_idx..=m11n_idx {
106            if needs_leap && new_moon_dates[i + 1] <= date_cst(ephemeris.solar_term[term]) {
107                months.push(NewMoon {
108                    month: Leap(month),
109                    date: new_moon_dates[i],
110                });
111                needs_leap = false;
112                continue;
113            }
114            month = month % 12 + 1;
115            months.push(NewMoon {
116                month: Common(month),
117                date: new_moon_dates[i],
118            });
119            term += 2;
120        }
121        assert!(!needs_leap);
122
123        Some(Annus {
124            annus,
125            ephemeris,
126            months,
127        })
128    }
129    /// 依特定日期取得其所在歲。
130    ///
131    /// 若曆表無該歲資料則返回 `None`。
132    ///
133    /// # 用例
134    ///
135    /// ```
136    /// use kalendarilo::Date;
137    /// use kalendarilo::chinese::Annus;
138    ///
139    /// let date = Date::from_gregorian(1999, 12, 31).unwrap();
140    /// let annus = Annus::from_date(date).unwrap();
141    ///
142    /// assert_eq!(2000, annus.annus);
143    /// ```
144    pub fn from_date(date: Date) -> Option<Self> {
145        let mut y = date.gregorian().0;
146        loop {
147            let annus = Self::new(y)?;
148
149            let start = annus.months[0].date;
150            let end = annus.months.last().unwrap().date;
151
152            if (start..end).contains(&date) {
153                return Some(annus);
154            }
155
156            y += if date < start { -1 } else { 1 };
157        }
158    }
159
160    /// 取得給定日期在該歲的年月日，返回格式為 `(年, 月, 日)`。
161    ///
162    /// 若所給日期不在該歲，則回報 `Err` 並指出該日期在該歲之前還是之後。
163    ///
164    /// # 用例
165    ///
166    /// ```
167    /// use kalendarilo::Date;
168    /// use kalendarilo::chinese::{Annus, Month::*};
169    ///
170    /// let date = Date::from_gregorian(2000, 1, 1).unwrap();
171    /// let annus = Annus::from_date(date).unwrap();
172    ///
173    /// assert_eq!(Ok((1999, Common(11), 25)), annus.ymd_for(date));
174    /// ```
175    pub fn ymd_for(&self, date: Date) -> Result<(i32, Month, i32), OtherAnnus> {
176        let begin = self.months[0].date;
177        let end = self.months.last().unwrap().date;
178
179        if date < begin {
180            return Err(OtherAnnus::Before);
181        } else if date >= end {
182            return Err(OtherAnnus::After);
183        }
184
185        let m = self
186            .months
187            .iter()
188            .take_while(|m| m.date <= date)
189            .last()
190            .unwrap();
191        let d = (date.jdn() + 1 - m.date.jdn()) as i32;
192        let y = if m.month.num() >= 11 {
193            self.annus - 1
194        } else {
195            self.annus
196        };
197        Ok((y, m.month, d))
198    }

200    /// 取得給定日期所在節氣信息，若當日並無交節，則給出該日相對其前一個交節的日數差。返回值格式如下：
201    ///
202    /// - `.0`：取得的節氣所在歲（前一歲大雪可能落在該歲，故須回報所在歲）
203    /// - `.1`：該節氣序號，1..=24 對應立春到大寒
204    /// - `.2`：所給 `date` 在該節交節後第幾日，為 0 則表示當日交節
205    ///
206    /// 本方法支持自該歲首日至次歲冬至前日的區間。
207    ///
208    /// 若給定日期不在該歲，或曆表無法取得前一歲節氣數據，則回報 `Err`。
209    ///
210    /// # 用例
211    ///
212    /// ```
213    /// use kalendarilo::Date;
214    /// use kalendarilo::chinese::{Annus, Month::*};
215    ///
216    /// let date = Date::from_gregorian(2000, 1, 1).unwrap();
217    /// let annus = Annus::from_date(date).unwrap();
218    ///
219    /// assert_eq!(Ok((2000, 22, 10)), annus.solar_term_for(date)); // 冬至過後第 10 天
220    /// ```
221    pub fn solar_term_for(&self, date: Date) -> Result<(i32, i32, i32), SolarTermErr> {
222        use self::OtherAnnus::*;
223        use SolarTermErr::*;
224        if date < self.months[0].date {
225            return Err(OtherAnnus(Before));
226        } else if date >= date_cst(self.ephemeris.solar_term[24]) {
227            return Err(OtherAnnus(After));
228        }
229        if date < date_cst(self.ephemeris.solar_term[0]) {
230            let last_annus = ephemeris::Annus::get(self.annus - 1).ok_or(NoData)?;
231            for (idx, &tdb) in (22..24).zip(&last_annus.solar_term[22..24]).rev() {
232                let term_start = date_cst(tdb);
233                if date >= term_start {
234                    return Ok((self.annus - 1, (idx + 21) % 24 + 1, date - term_start));
235                }
236            }
237            panic!("incorrect data for annus {}", self.annus - 1);
238        }
239        let idx = self.ephemeris.solar_term[..24].partition_point(|&tdb| date_cst(tdb) <= date) - 1;
240        let off = date - date_cst(self.ephemeris.solar_term[idx]);
241        Ok((self.annus, (idx as i32 + 21) % 24 + 1, off))
242    }
243}

245/// 表示給定日期不在該歲，並指出其在前還是在後。
246#[derive(Debug, Copy, Clone, Eq, PartialEq)]
247pub enum OtherAnnus {
248    Before,
249    After,
250}
251
252/// 表示給定日期不在該歲，或曆表無法取得節氣數據。
253#[derive(Debug, Copy, Clone, Eq, PartialEq)]
254pub enum SolarTermErr {
255    NoData,
256    OtherAnnus(OtherAnnus),
257}
258
259/// 將給定曆表時間轉為北京時間（UTC+8）日期。
260pub fn date_cst(tdb: Tdb) -> Date {
261    Ut::convert(tdb).date_in_timezone(480)
262}

264/// 取得所給公元年的干支。
265///
266/// # 用例
267///
268/// ```
269/// use kalendarilo::chinese::sexagenary_for_year;
270///
271/// assert_eq!(1, sexagenary_for_year(-2696));
272/// ```
273pub fn sexagenary_for_year(year: i32) -> i32 {
274    (year + 2696).rem_euclid(60) + 1
275}
