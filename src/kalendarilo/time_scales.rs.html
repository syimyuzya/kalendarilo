13/// [Barycentric dynamic time](https://en.wikipedia.org/wiki/Barycentric_Dynamical_Time),
14/// represented in Julian date (JD).
15///
16/// Ephemeris data are typically computed in this
17/// time scale, and should be converted to UT when calculating dates.
18#[derive(Debug, Copy, Clone, PartialEq, PartialOrd)]
19pub struct Tdb(pub f64);
20
21/// [Terristrial time](https://en.wikipedia.org/wiki/Terrestrial_Time),
22/// represented in Julian date (JD).
23///
24/// Note: Because TT differs no more than centisenconds from TDB during
25/// thousands of years, they are treated numerically the same in this crate for
26/// calendar calculation.
27#[derive(Debug, Copy, Clone, PartialEq, PartialOrd)]
28pub struct Tt(pub f64);
29
30impl From<Tdb> for Tt {
31    fn from(tdb: Tdb) -> Tt {
32        Tt(tdb.0)
33    }
34}
35
36impl From<Tai> for Tt {
37    fn from(tai: Tai) -> Tt {
38        Tt(tai.0 + 32.184 / 86400.0)
39    }
40}
41
42/// [International atomic time](https://en.wikipedia.org/wiki/International_Atomic_Time),
43/// represented in Julian date (JD).
44#[derive(Debug, Copy, Clone, PartialEq, PartialOrd)]
45pub struct Tai(pub f64);
46
47impl From<Tt> for Tai {
48    fn from(tt: Tt) -> Tai {
49        Tai(tt.0 - 32.184 / 86400.0)
50    }
51}
52
53impl From<Tdb> for Tai {
54    fn from(tdb: Tdb) -> Tai {
55        Tt::from(tdb).into()
56    }
57}

59/// [Universal time](https://en.wikipedia.org/wiki/Universal_Time), the actual
60/// civil time used for determining the actual date at a given time point.
61///
62/// This can be either UTC (coordinated universal time, based on TAI with leap
63/// seconds) or UT1 (mean solar time on the Prime Meridian) depending on the
64/// year. Specifically, UTC from 1972-01-01T00:00Z to the latest known leap
65/// second, UT1 otherwise.
66///
67/// Due to irregularity of Earth's rotation, conversion from TAI to UT1 relies
68/// on data points with inter-/extrapolation with
69/// [a method described here](https://astro.ukho.gov.uk/nao/lvm/).
70#[derive(Debug, Copy, Clone, PartialEq, PartialOrd)]
71pub struct Ut(pub f64);
72
73impl Ut {
74    // XXX Liveri `Enum`-n por distingi inter UTC & UT1
75    /// Tries to convert a TAI (or other time scale easily convertible to TAI)
76    /// into UT.
77    ///
78    /// # Example
79    ///
80    /// ```
81    /// use kalendarilo::time_scales::{Tdb, Ut};
82    /// let tdb = Tdb(2451543.166666667);
83    /// let ut = Ut::convert(tdb);
84    /// assert_eq!((1999, 12, 30), ut.date_in_timezone(0).gregorian());
85    /// ```
86    ///
87    /// # Panics
88    ///
89    /// Does not currently support time before 1972-01-01 and will panic.
90    /// Working on it.
91    pub fn convert<T>(time: T) -> Self
92    where
93        T: Into<Tai>,
94    {
95        let tai = time.into();
96        let &leap_seconds::Data {
97            starts,
98            ref leap_seconds,
99            expires,
100            c2,
101        } = leap_seconds::data();
102
103        if tai < starts {
104            todo!("UT before UTC (1972-01-01)");
105        } else if tai > expires {
106            let diff = leap_seconds::estimate(tai) + c2;
107            return Ut(tai.0 - diff / 86400.0); // NOTE UT1, ne UTC
108        }
109
110        let ls = match leap_seconds.partition_point(|ls| ls.tai <= tai) {
111            0 => return Ut(tai.0 - 10.0 / 86400.0),
112            i => &leap_seconds[i - 1],
113        };
114        let leap = (tai.0 - ls.tai.0).min(2.0) / 2.0;
115        Ut(tai.0 - (ls.delta_secs as f64 + leap) / 86400.0)
116    }
117    /// Returns the date at the time point in timezone ahead (east) of UTC by
118    /// `tz_offset_minutes`minutes.
119    ///
120    /// For Beijing time (UTC+8), `tz_offset_minutes` should be +480.
121    ///
122    /// # Example
123    ///
124    /// ```
125    /// use kalendarilo::time_scales::{Tdb, Ut};
126    /// let tdb = Tdb(2451543.166666667);
127    /// let ut = Ut::convert(tdb);
128    /// assert_eq!((1999, 12, 30), ut.date_in_timezone(480).gregorian());
129    /// ```
130    pub fn date_in_timezone(&self, tz_offset_minutes: i32) -> Date {
131        let jdn = (self.0 + tz_offset_minutes as f64 / 1440.0).round() as u32;
132        Date::from_jdn(jdn)
133    }
134}

136mod leap_seconds {
137    use super::{Tai, Tt};
138    use crate::date::Date;
139    use std::sync::LazyLock;
140
141    pub const DATES: &[(i32, i32, i32)] = &[
142        (1972, 6, 30),
143        (1972, 12, 31),
144        (1973, 12, 31),
145        (1974, 12, 31),
146        (1975, 12, 31),
147        (1976, 12, 31),
148        (1977, 12, 31),
149        (1978, 12, 31),
150        (1979, 12, 31),
151        (1981, 6, 30),
152        (1982, 6, 30),
153        (1983, 6, 30),
154        (1985, 6, 30),
155        (1987, 12, 31),
156        (1989, 12, 31),
157        (1990, 12, 31),
158        (1992, 6, 30),
159        (1993, 6, 30),
160        (1994, 6, 30),
161        (1995, 12, 31),
162        (1997, 6, 30),
163        (1998, 12, 31),
164        (2005, 12, 31),
165        (2008, 12, 31),
166        (2012, 6, 30),
167        (2015, 6, 30),
168        (2016, 12, 31),
169    ];
170    pub const DATE_EXPIRES: (i32, i32, i32) = (2021, 12, 31);
171
172    #[derive(Debug, Clone)]
173    pub struct Data {
174        pub starts: Tai,
175        pub leap_seconds: Vec<LeapSecond>,
176        pub expires: Tai,
177        pub c2: f64,
178    }
179    #[derive(Debug, Clone)]
180    pub struct LeapSecond {
181        pub tai: Tai,
182        pub delta_secs: i32,
183    }
184
185    static DATA: LazyLock<Data> = LazyLock::new(|| {
186        let starts = Tai(Date::from_gregorian(1972, 1, 1).unwrap().jdn() as f64 + 10.0 / 86400.0);
187        let mut leap_seconds = Vec::with_capacity(DATES.len());
188        for (delta_secs, &(y, m, d)) in (10..).zip(DATES) {
189            let jdn = Date::from_gregorian(y, m, d)
190                .unwrap_or_else(|| panic!("date not recognized: {:?}", (y, m, d)))
191                .jdn();
192            let tai = Tai(jdn as f64 + (43199 + delta_secs) as f64 / 86400.0);
193            leap_seconds.push(LeapSecond { tai, delta_secs });
194        }
195        let (y, m, d) = DATE_EXPIRES;
196        let jdn = Date::from_gregorian(y, m, d)
197            .unwrap_or_else(|| panic!("date not recognized: {:?}", (y, m, d)))
198            .jdn();
199        let tai = Tai(jdn as f64 + (43199 + 10 + DATES.len()) as f64 / 86400.0);
200        let c2 = (DATES.len() + 10) as f64 - estimate(tai);
201        Data {
202            starts,
203            leap_seconds,
204            expires: tai,
205            c2,
206        }
207    });
208
209    pub fn data() -> &'static Data {
210        &DATA
211    }
212
213    pub fn estimate<T: Into<Tt>>(tt: T) -> f64 {
214        use std::f64::consts::PI;
215        let tt = tt.into();
216        let y = (tt.0 - 2451544.5) / 365.2425 + 2000.0;
217        let t = (y - 1825.0) / 100.0;
218        31.4115 * t * t + 284.8435805251424 * (2.0 * PI * (t + 0.75) / 14.0).cos()
219    }
220}

222#[cfg(test)]
223mod tests {
224    use super::*;
225
226    #[test]
227    fn tdb_to_ut() {
228        let tdb = Tdb(2451543.166666667);
229        let ut = Ut::convert(tdb);
230        assert_eq!((1999, 12, 30), ut.date_in_timezone(0).gregorian());
231        assert_eq!((1999, 12, 30), ut.date_in_timezone(480).gregorian());
232        let ut_midnight = Ut(ut.0 + (32.0 + 32.184) / 86400.0);
233        let ut_before_midnight = Ut(ut_midnight.0 - 1.0 / 86400.0);
234        assert_eq!(
235            (1999, 12, 30),
236            ut_before_midnight.date_in_timezone(480).gregorian()
237        );
238        assert_eq!(
239            (1999, 12, 31),
240            ut_midnight.date_in_timezone(480).gregorian()
241        );
242    }
243
244    #[test]
245    fn playing_with() {
246        let tdb = Tdb(2462501.166666667 + 5.647029454550371); // 2030 小寒
247        let ut = Ut::convert(tdb);
248        assert!((ut.0 - 2462506.81319).abs() <= 30.0 / 86400.0);
249    }
